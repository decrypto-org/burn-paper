\begin{algorithm}[H]
    \caption{\label{alg.verify-event-federation} A \textsf{verify} implementation which verifies a spend event $e$ on the source blockchain using the federation mechanism. $\mathcal{M}$ denotes the latest MMR approved by the federation majority.}
    \begin{algorithmic}[1]
        \Function{$\mathsf{verify}_\mathcal{M}^{e}$}{$\alpha, \textsf{blockConnection}$}
            \If{$\neg \textsf{verify-tx}_e(\alpha)$}
                \State\Return{$\false$}
            \EndIf
            \Let{\textsf{blockTreeRoot}}{\textsf{calculateMerkleRoot}(id(\alpha.\textsf{blockHeader}), \textsf{blockConnection})}
            \If{$\textsf{blockTreeRoot} = \mathcal{M} \land \textsf{txMerkleRoot} = blk.\textsf{txMerkleRoot}$}
                \State\Return{$\true$}
            \EndIf
        \EndFunction
        \vskip8pt
    \end{algorithmic}
\end{algorithm}
