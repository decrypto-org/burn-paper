\begin{algorithm}[H]
    \caption{\label{alg.verify-event-federation} A \textsf{verify} implementation which verifies a spend event $e$ on the source blockchain using the federation mechanism. $\mathcal{M}$ denotes the latest MMR approved by the federation majority.}
    \begin{algorithmic}[1]
        \Function{$\mathsf{verify}_\mathcal{M}$}{$\textsf{addr}, \textsf{amount}, b, \textsf{tx}, \tau, \textsf{blockConnection}$}
            \If{$\neg \textsf{verify-tx}(\textsf{addr}, \textsf{amount}, b, \textsf{tx}, \tau)$}
                \State\Return{$\false$}
            \EndIf
            \State\Return{$\VerMT(\mathcal{M}, b, \mathsf{blockConnection})$}
        \EndFunction
        \vskip8pt
    \end{algorithmic}
\end{algorithm}
