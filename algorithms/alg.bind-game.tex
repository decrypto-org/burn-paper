\begin{algorithm}[H]
    \caption{\label{alg.bind-game} The challenger for the burn protocol game-based security.}
    \begin{algorithmic}[1]
        \Function{\bindattack$_{\mathcal{A},\Pi}$}{$\kappa$}
            \Let{(t, t', \burnAddr)}{\mathcal{A}(1^\kappa)}
            \If{$\lnot \BurnVerify(1^\kappa, t, \burnAddr)$}
                \State\Return$\false$
            \EndIf
            \If{$\lnot \BurnVerify(1^\kappa, t', \burnAddr)$}
                \State\Return$\false$
            \EndIf
            \State\Return$(t \neq t')$
        \EndFunction
    \end{algorithmic}
\end{algorithm}
