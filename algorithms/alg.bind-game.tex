\begin{algorithm}[H]
    \caption{\label{alg.bind-game} The challenger for the burn protocol game-based security.}
    \begin{algorithmic}[1]
        \Function{\textsc{bind-attack}$_{\mathcal{A},\Pi}$}{$\lambda$}
            \Let{(t, t', \mathsf{burnAddr})}{\mathcal{A}(1^\lambda)}
            \If{$t \neq t' \land \textsf{BurnVerify(t, burnAddr)} \land \textsf{BurnVerify}(t', burnAddr)$}
                \State\Return{\sf true}
            \Else
                \State\Return{\sf false}
            \EndIf
        \EndFunction
    \end{algorithmic}
\end{algorithm}
