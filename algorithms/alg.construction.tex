\begin{algorithm}[H]
    \caption{\label{alg.construction} Our uncensorable proof-of-burn protocol for Bitcoin P2PKH.}
    \begin{algorithmic}[1]
        \Function{$\GenBurnAddress_H$}{$1^\kappa, t$}
            \Let{th}{H_\kappa(t)}
            \Let{th'}{th \xor 1}
            \Comment{Key perturbation}
            \State\Return{th'}
        \EndFunction
        \Function{$\BurnVerify_H$}{$1^\kappa, t, th'$}
            \State\Return{$(\GenBurnAddress_H(1^\kappa, t) = th')$}
        \EndFunction
        \vskip8pt
    \end{algorithmic}
\end{algorithm}
