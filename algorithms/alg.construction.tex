\begin{algorithm}[H]
    \caption{\label{alg.construction} Our Uncensorable Proof-of-Burn protocol for Bitcoin p2pkh.}
    \begin{algorithmic}[1]
        \Function{\textsf{GenBurnAddress}$_H$}{$1^\kappa, t$}
            \Let{th}{H_\kappa(t)}
            \Let{th'}{th \oplus 1}
            \Comment{Key perturbation}
            \State\Return{th'}
        \EndFunction
        \Function{\textsf{BurnVerify}$_H$}{$1^\kappa, t, th'$}
            \If{$\textsf{GenBurnAddress}_H(1^\kappa, t) = th'$}
                \State\Return\textsf{true}
            \Else
                \State\Return\textsf{false}
            \EndIf
        \EndFunction
        \vskip8pt
    \end{algorithmic}
\end{algorithm}
