\begin{algorithm}[H]
    \caption{\label{alg.verify-tx} A function to verify that a transaction is conforming to the provided event and is included in the provided block.}
    \begin{algorithmic}[1]
        \Function{\sf verify-tx$_e$}{$\alpha$}
            \If{$\neg \textsf{containsTransfer}(a.\mathsf{tx}, e.\textsf{receivingAddress}, e.\textsf{amount})$}
                \State\Return{$\false$}
            \EndIf
            \Let{\textsf{txMerkleRoot}}{\textsf{calculateMerkleRoot}(H(\alpha.\textsf{tx}), \alpha.\textsf{txInclusion})}
            \If{$\textsf{txMerkleRoot} \neq a.\textsf{blk.txMerkleRoot}$}
                \State\Return{$\false$}
            \EndIf

            \State\Return{$\true$}
        \EndFunction
        \vskip8pt
    \end{algorithmic}
\end{algorithm}
