\begin{algorithm}[H]
    \caption{\label{alg.verify-tx} A function to verify that a transaction is conforming to the provided event and is included in the provided block.}
    \begin{algorithmic}[1]
        \Function{\sf verify-tx$_e$}{$\alpha$}
            \If{$\neg \TxHasTransfer(a.\mathsf{tx}, e.\textsf{receivingAddress}, e.\textsf{amount})$}
                \State\Return{$\false$}
            \EndIf
            \State\Return{$\VerTxIncl(a.b, \alpha.\textsf{tx}, \alpha.\tau)$}
        \EndFunction
        \vskip8pt
    \end{algorithmic}
\end{algorithm}
