\begin{algorithm}[H]
    \caption{\label{alg.construction-real} The key perturbation algorithm which
    generates a provable proof-of-burn address.}
    \begin{algorithmic}[1]
        \Function{\textsf{GenBurnAddress}}{$1^\kappa, t$}
            \Let{pkr}{\textsf{SHA512}(t)}
            \Let{pkh}{\textsf{RIPEMD160}(\textsf{SHA256}(\texttt{0x04} || pkr)}
            \Let{pkh'}{pkh \xor \texttt{0x01}}
            \Comment{Key perturbation}
            \If{main net}
              \Let{addr}{\texttt{0x00} || pkh'}
            \Else
              \Let{addr}{\texttt{0x6f} || pkh'}
            \EndIf
            \Let{checksum}{\textsf{SHA256}(\textsf{SHA256}(addr))[4 * 8]}
            \State\Return{\textsf{base58}(addr || checksum)}
        \EndFunction
        \vskip8pt
    \end{algorithmic}
\end{algorithm}
\end{figure}
