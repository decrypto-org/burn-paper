\begin{algorithm}[H]
    \caption{\label{alg.key-predict-game} The challenger for the public key prediction adversary.}
    \begin{algorithmic}[1]
        \Function{\textsc{pk-predict}$_{\mathcal{A},S}$}{$\kappa$}
            \Let{pk}{\mathcal{A}(1^\kappa)}
            \Let{pk'}{\textsf{Gen}(1^\kappa)}
            \If{$pk = pk'$}
                \State\Return{\sf true}
            \Else
                \State\Return{\sf false}
            \EndIf
        \EndFunction
    \end{algorithmic}
\end{algorithm}
