\begin{algorithm}[H]
    \caption{\label{alg.perturbation} The key perturbation algorithm which
             generates a provable proof-of-burn address.}
    \begin{algorithmic}[1]
        \Function{\textsf{GenBurnAddress}$_H$}{$1^\kappa, t$}
            \Let{th}{H_\kappa(t)}
            \Let{th'}{th \oplus 1}
            \Comment{Key perturbation}
            \State\Return{th'}
        \EndFunction
        \Function{\textsf{BurnVerify}$_H$}{$1^\kappa, t, th'$}
            \If{$\textsf{GenBurnAdress}_H(1^\kappa, t) = th'$}
            \EndIf
        \EndFunction
        \vskip8pt
    \end{algorithmic}
\end{algorithm}
